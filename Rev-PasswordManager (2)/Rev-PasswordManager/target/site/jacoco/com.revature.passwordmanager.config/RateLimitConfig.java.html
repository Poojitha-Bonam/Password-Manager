@Configuration
@ConfigurationProperties(prefix = "rate-limit")
public class RateLimitConfig {

  private int generalRequestsPerMinute = 60;
  private int authRequestsPerMinute = 10;
  private int passwordGeneratorRequestsPerMinute = 30;
